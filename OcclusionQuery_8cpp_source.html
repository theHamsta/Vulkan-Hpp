30 int main( int /*argc*/, char ** /*argv*/ )

34  vk::Instance instance = vk::su::createInstance( AppName, EngineName, {}, vk::su::getInstanceExtensions() );
35 #if !defined( NDEBUG )
36  vk::DebugUtilsMessengerEXT debugUtilsMessenger = instance.createDebugUtilsMessengerEXT( vk::su::makeDebugUtilsMessengerCreateInfoEXT() );
37 #endif
38 
39  vk::PhysicalDevice physicalDevice = instance.enumeratePhysicalDevices().front();
40 
41  vk::su::SurfaceData surfaceData( instance, AppName, vk::Extent2D( 500, 500 ) );
42 
43  std::pair<uint32_t, uint32_t> graphicsAndPresentQueueFamilyIndex = vk::su::findGraphicsAndPresentQueueFamilyIndex( physicalDevice, surfaceData.surface );
44  vk::Device device = vk::su::createDevice( physicalDevice, graphicsAndPresentQueueFamilyIndex.first, vk::su::getDeviceExtensions() );
45 
46  vk::CommandPool commandPool = device.createCommandPool( { {}, graphicsAndPresentQueueFamilyIndex.first } );
47  vk::CommandBuffer commandBuffer =
48  device.allocateCommandBuffers( vk::CommandBufferAllocateInfo( commandPool, vk::CommandBufferLevel::ePrimary, 1 ) ).front();
49 
50  vk::Queue graphicsQueue = device.getQueue( graphicsAndPresentQueueFamilyIndex.first, 0 );
51  vk::Queue presentQueue = device.getQueue( graphicsAndPresentQueueFamilyIndex.second, 0 );
52 
53  vk::su::SwapChainData swapChainData( physicalDevice,
54  device,
55  surfaceData.surface,
56  surfaceData.extent,
57  vk::ImageUsageFlagBits::eColorAttachment | vk::ImageUsageFlagBits::eTransferSrc,
58  {},
59  graphicsAndPresentQueueFamilyIndex.first,
60  graphicsAndPresentQueueFamilyIndex.second );
61 
62  vk::su::DepthBufferData depthBufferData( physicalDevice, device, vk::Format::eD16Unorm, surfaceData.extent );
63 
64  vk::su::BufferData uniformBufferData( physicalDevice, device, sizeof( glm::mat4x4 ), vk::BufferUsageFlagBits::eUniformBuffer );
65  glm::mat4x4 mvpcMatrix = vk::su::createModelViewProjectionClipMatrix( surfaceData.extent );
66  vk::su::copyToDevice( device, uniformBufferData.deviceMemory, mvpcMatrix );
67 
68  vk::DescriptorSetLayout descriptorSetLayout =
69  vk::su::createDescriptorSetLayout( device, { { vk::DescriptorType::eUniformBuffer, 1, vk::ShaderStageFlagBits::eVertex } } );
70  vk::PipelineLayout pipelineLayout = device.createPipelineLayout( vk::PipelineLayoutCreateInfo( vk::PipelineLayoutCreateFlags(), descriptorSetLayout ) );
71 
72  vk::RenderPass renderPass = vk::su::createRenderPass(
73  device, vk::su::pickSurfaceFormat( physicalDevice.getSurfaceFormatsKHR( surfaceData.surface ) ).format, depthBufferData.format );
74 
75  glslang::InitializeProcess();
76  vk::ShaderModule vertexShaderModule = vk::su::createShaderModule( device, vk::ShaderStageFlagBits::eVertex, vertexShaderText_PC_C );
77  vk::ShaderModule fragmentShaderModule = vk::su::createShaderModule( device, vk::ShaderStageFlagBits::eFragment, fragmentShaderText_C_C );
78  glslang::FinalizeProcess();
79 
80  std::vector<vk::Framebuffer> framebuffers =
81  vk::su::createFramebuffers( device, renderPass, swapChainData.imageViews, depthBufferData.imageView, surfaceData.extent );
82 
83  vk::su::BufferData vertexBufferData( physicalDevice, device, sizeof( coloredCubeData ), vk::BufferUsageFlagBits::eVertexBuffer );
84  vk::su::copyToDevice( device, vertexBufferData.deviceMemory, coloredCubeData, sizeof( coloredCubeData ) / sizeof( coloredCubeData[0] ) );
85 
86  vk::DescriptorPool descriptorPool = vk::su::createDescriptorPool( device, { { vk::DescriptorType::eUniformBuffer, 1 } } );
87  vk::DescriptorSetAllocateInfo descriptorSetAllocateInfo( descriptorPool, descriptorSetLayout );
88  vk::DescriptorSet descriptorSet = device.allocateDescriptorSets( descriptorSetAllocateInfo ).front();
89 
90  vk::su::updateDescriptorSets( device, descriptorSet, { { vk::DescriptorType::eUniformBuffer, uniformBufferData.buffer, VK_WHOLE_SIZE, {} } }, {} );
91 
92  vk::PipelineCache pipelineCache = device.createPipelineCache( vk::PipelineCacheCreateInfo() );
93  vk::Pipeline graphicsPipeline = vk::su::createGraphicsPipeline( device,
94  pipelineCache,
95  std::make_pair( vertexShaderModule, nullptr ),
96  std::make_pair( fragmentShaderModule, nullptr ),
97  sizeof( coloredCubeData[0] ),
98  { { vk::Format::eR32G32B32A32Sfloat, 0 }, { vk::Format::eR32G32B32A32Sfloat, 16 } },
99  vk::FrontFace::eClockwise,
100  true,
101  pipelineLayout,
102  renderPass );
103 
104  /* VULKAN_KEY_START */
105 
106  vk::Semaphore imageAcquiredSemaphore = device.createSemaphore( vk::SemaphoreCreateInfo( vk::SemaphoreCreateFlags() ) );
107 
108  // Get the index of the next available swapchain image:
109  vk::ResultValue<uint32_t> currentBuffer = device.acquireNextImageKHR( swapChainData.swapChain, UINT64_MAX, imageAcquiredSemaphore, nullptr );
110  assert( currentBuffer.result == vk::Result::eSuccess );
111  assert( currentBuffer.value < framebuffers.size() );
112 
113  /* Allocate a uniform buffer that will take query results. */
114  vk::Buffer queryResultBuffer = device.createBuffer( vk::BufferCreateInfo(
115  vk::BufferCreateFlags(), 4 * sizeof( uint64_t ), vk::BufferUsageFlagBits::eUniformBuffer | vk::BufferUsageFlagBits::eTransferDst ) );
116 
117  vk::MemoryRequirements memoryRequirements = device.getBufferMemoryRequirements( queryResultBuffer );
118  uint32_t memoryTypeIndex = vk::su::findMemoryType( physicalDevice.getMemoryProperties(),
119  memoryRequirements.memoryTypeBits,
120  vk::MemoryPropertyFlagBits::eHostVisible | vk::MemoryPropertyFlagBits::eHostCoherent );
121  vk::DeviceMemory queryResultMemory = device.allocateMemory( vk::MemoryAllocateInfo( memoryRequirements.size, memoryTypeIndex ) );
122 
123  device.bindBufferMemory( queryResultBuffer, queryResultMemory, 0 );
124 
125  vk::QueryPool queryPool =
126  device.createQueryPool( vk::QueryPoolCreateInfo( vk::QueryPoolCreateFlags(), vk::QueryType::eOcclusion, 2, vk::QueryPipelineStatisticFlags() ) );
127 
128  commandBuffer.begin( vk::CommandBufferBeginInfo( vk::CommandBufferUsageFlags() ) );
129  commandBuffer.resetQueryPool( queryPool, 0, 2 );
130 
131  std::array<vk::ClearValue, 2> clearValues;
132  clearValues[0].color = vk::ClearColorValue( 0.2f, 0.2f, 0.2f, 0.2f );
133  clearValues[1].depthStencil = vk::ClearDepthStencilValue( 1.0f, 0 );
134  commandBuffer.beginRenderPass(
135  vk::RenderPassBeginInfo( renderPass, framebuffers[currentBuffer.value], vk::Rect2D( vk::Offset2D(), surfaceData.extent ), clearValues ),
136  vk::SubpassContents::eInline );
137 
138  commandBuffer.bindPipeline( vk::PipelineBindPoint::eGraphics, graphicsPipeline );
139  commandBuffer.bindDescriptorSets( vk::PipelineBindPoint::eGraphics, pipelineLayout, 0, descriptorSet, {} );
140 
141  commandBuffer.bindVertexBuffers( 0, vertexBufferData.buffer, { 0 } );
142  commandBuffer.setViewport(
143  0, vk::Viewport( 0.0f, 0.0f, static_cast<float>( surfaceData.extent.width ), static_cast<float>( surfaceData.extent.height ), 0.0f, 1.0f ) );
144  commandBuffer.setScissor( 0, vk::Rect2D( vk::Offset2D( 0, 0 ), surfaceData.extent ) );
145 
146  commandBuffer.beginQuery( queryPool, 0, vk::QueryControlFlags() );
147  commandBuffer.endQuery( queryPool, 0 );
148 
149  commandBuffer.beginQuery( queryPool, 1, vk::QueryControlFlags() );
150  commandBuffer.draw( 12 * 3, 1, 0, 0 );
151  commandBuffer.endRenderPass();
152  commandBuffer.endQuery( queryPool, 1 );
153 
154  commandBuffer.copyQueryPoolResults(
155  queryPool, 0, 2, queryResultBuffer, 0, sizeof( uint64_t ), vk::QueryResultFlagBits::e64 | vk::QueryResultFlagBits::eWait );
156  commandBuffer.end();
157 
158  vk::Fence drawFence = device.createFence( vk::FenceCreateInfo() );
159 
160  vk::PipelineStageFlags waitDestinationStageMask( vk::PipelineStageFlagBits::eColorAttachmentOutput );
161  vk::SubmitInfo submitInfo( imageAcquiredSemaphore, waitDestinationStageMask, commandBuffer );
162  graphicsQueue.submit( submitInfo, drawFence );
163 
164  graphicsQueue.waitIdle();
165 
166  vk::ResultValue<std::vector<uint64_t>> rv = device.getQueryPoolResults<uint64_t>(
167  queryPool, 0, 2, 2 * sizeof( uint64_t ), sizeof( uint64_t ), vk::QueryResultFlagBits::e64 | vk::QueryResultFlagBits::eWait );
168  switch ( rv.result )
169  {
170  case vk::Result::eSuccess: break;
171  case vk::Result::eNotReady: std::cout << "vk::Device::getQueryPoolResults returned vk::Result::eNotReady !\n"; break;
172  default: assert( false ); // an unexpected result is returned !
173  }
174 
175  std::cout << "vkGetQueryPoolResults data\n";
176  std::cout << "samples_passed[0] = " << rv.value[0] << "\n";
177  std::cout << "samples_passed[1] = " << rv.value[1] << "\n";
178 
179  /* Read back query result from buffer */
180  uint64_t * samplesPassedPtr = static_cast<uint64_t *>( device.mapMemory( queryResultMemory, 0, memoryRequirements.size, vk::MemoryMapFlags() ) );
181 
182  std::cout << "vkCmdCopyQueryPoolResults data\n";
183  std::cout << "samples_passed[0] = " << samplesPassedPtr[0] << "\n";
184  std::cout << "samples_passed[1] = " << samplesPassedPtr[1] << "\n";
185 
186  device.unmapMemory( queryResultMemory );
187 
188  while ( vk::Result::eTimeout == device.waitForFences( drawFence, VK_TRUE, vk::su::FenceTimeout ) )
189  ;
190 
191  vk::Result result = presentQueue.presentKHR( vk::PresentInfoKHR( {}, swapChainData.swapChain, currentBuffer.value ) );
192  switch ( result )
193  {
194  case vk::Result::eSuccess: break;
195  case vk::Result::eSuboptimalKHR: std::cout << "vk::Queue::presentKHR returned vk::Result::eSuboptimalKHR !\n"; break;
196  default: assert( false ); // an unexpected result is returned !
197  }
198  std::this_thread::sleep_for( std::chrono::milliseconds( 1000 ) );
199 
200  device.destroyFence( drawFence );
201  device.destroyQueryPool( queryPool );
202  device.destroyBuffer( queryResultBuffer ); // destroy the queryResultBuffer before freeing the bound queryResultMemory !
203  device.freeMemory( queryResultMemory );
204  device.destroySemaphore( imageAcquiredSemaphore );
205 
206  /* VULKAN_KEY_END */
207 
208  device.destroyPipeline( graphicsPipeline );
209  device.destroyPipelineCache( pipelineCache );
210  device.freeDescriptorSets( descriptorPool, descriptorSet );
211  device.destroyDescriptorPool( descriptorPool );
212  vertexBufferData.clear( device );
213  for ( auto framebuffer : framebuffers )
214  {
215  device.destroyFramebuffer( framebuffer );
216  }
217  device.destroyShaderModule( fragmentShaderModule );
218  device.destroyShaderModule( vertexShaderModule );
219  device.destroyRenderPass( renderPass );
220  device.destroyPipelineLayout( pipelineLayout );
221  device.destroyDescriptorSetLayout( descriptorSetLayout );
222  uniformBufferData.clear( device );
223  depthBufferData.clear( device );
224  swapChainData.clear( device );
225  device.freeCommandBuffers( commandPool, commandBuffer );
226  device.destroyCommandPool( commandPool );
227  device.destroy();
228  instance.destroySurfaceKHR( surfaceData.surface );
229 #if !defined( NDEBUG )
230  instance.destroyDebugUtilsMessengerEXT( debugUtilsMessenger );
231 #endif
232  instance.destroy();
233  }
234  catch ( vk::SystemError & err )
235  {
236  std::cout << "vk::SystemError: " << err.what() << std::endl;
237  exit( -1 );
238  }
239  catch ( std::exception & err )
240  {
241  std::cout << "std::exception: " << err.what() << std::endl;
242  exit( -1 );
243  }
244  catch ( ... )
245  {
246  std::cout << "unknown error\n";
247  exit( -1 );
248  }
249  return 0;
250 }
