23 static char const * AppName = "13_InitVertexBuffer";
24 static char const * EngineName = "Vulkan.hpp";

26 int main( int /*argc*/, char ** /*argv*/ )
27 {
28  try
29  {
30  vk::Instance instance = vk::su::createInstance( AppName, EngineName, {}, vk::su::getInstanceExtensions() );
31 #if !defined( NDEBUG )
32  vk::DebugUtilsMessengerEXT debugUtilsMessenger = instance.createDebugUtilsMessengerEXT( vk::su::makeDebugUtilsMessengerCreateInfoEXT() );
33 #endif
34 
35  vk::PhysicalDevice physicalDevice = instance.enumeratePhysicalDevices().front();
36 
37  vk::su::SurfaceData surfaceData( instance, AppName, vk::Extent2D( 64, 64 ) );
38 
39  std::pair<uint32_t, uint32_t> graphicsAndPresentQueueFamilyIndex = vk::su::findGraphicsAndPresentQueueFamilyIndex( physicalDevice, surfaceData.surface );
40  vk::Device device = vk::su::createDevice( physicalDevice, graphicsAndPresentQueueFamilyIndex.first, vk::su::getDeviceExtensions() );
41 
42  vk::CommandPool commandPool = device.createCommandPool( { {}, graphicsAndPresentQueueFamilyIndex.first } );
43  vk::CommandBuffer commandBuffer =
44  device.allocateCommandBuffers( vk::CommandBufferAllocateInfo( commandPool, vk::CommandBufferLevel::ePrimary, 1 ) ).front();
45 
46  vk::Queue graphicsQueue = device.getQueue( graphicsAndPresentQueueFamilyIndex.first, 0 );
47 
48  vk::su::SwapChainData swapChainData( physicalDevice,
49  device,
50  surfaceData.surface,
51  surfaceData.extent,
52  vk::ImageUsageFlagBits::eColorAttachment | vk::ImageUsageFlagBits::eTransferSrc,
53  {},
54  graphicsAndPresentQueueFamilyIndex.first,
55  graphicsAndPresentQueueFamilyIndex.second );
56 
57  vk::su::DepthBufferData depthBufferData( physicalDevice, device, vk::Format::eD16Unorm, surfaceData.extent );
58 
59  vk::RenderPass renderPass = vk::su::createRenderPass( device, swapChainData.colorFormat, depthBufferData.format );
60 
61  std::vector<vk::Framebuffer> framebuffers =
62  vk::su::createFramebuffers( device, renderPass, swapChainData.imageViews, depthBufferData.imageView, surfaceData.extent );
63 
64  /* VULKAN_KEY_START */
65 
66  // create a vertex buffer for some vertex and color data
67  vk::Buffer vertexBuffer =
68  device.createBuffer( vk::BufferCreateInfo( vk::BufferCreateFlags(), sizeof( coloredCubeData ), vk::BufferUsageFlagBits::eVertexBuffer ) );
69 
70  // allocate device memory for that buffer
71  vk::MemoryRequirements memoryRequirements = device.getBufferMemoryRequirements( vertexBuffer );
72  uint32_t memoryTypeIndex = vk::su::findMemoryType( physicalDevice.getMemoryProperties(),
73  memoryRequirements.memoryTypeBits,
74  vk::MemoryPropertyFlagBits::eHostVisible | vk::MemoryPropertyFlagBits::eHostCoherent );
75  vk::DeviceMemory deviceMemory = device.allocateMemory( vk::MemoryAllocateInfo( memoryRequirements.size, memoryTypeIndex ) );
76 
77  // copy the vertex and color data into that device memory
78  uint8_t * pData = static_cast<uint8_t *>( device.mapMemory( deviceMemory, 0, memoryRequirements.size ) );
79  memcpy( pData, coloredCubeData, sizeof( coloredCubeData ) );
80  device.unmapMemory( deviceMemory );
81 
82  // and bind the device memory to the vertex buffer
83  device.bindBufferMemory( vertexBuffer, deviceMemory, 0 );
84 
85  vk::Semaphore imageAcquiredSemaphore = device.createSemaphore( vk::SemaphoreCreateInfo( vk::SemaphoreCreateFlags() ) );
86  vk::ResultValue<uint32_t> currentBuffer = device.acquireNextImageKHR( swapChainData.swapChain, vk::su::FenceTimeout, imageAcquiredSemaphore, nullptr );
87  assert( currentBuffer.result == vk::Result::eSuccess );
88  assert( currentBuffer.value < framebuffers.size() );
89 
90  std::array<vk::ClearValue, 2> clearValues;
91  clearValues[0].color = vk::ClearColorValue( 0.2f, 0.2f, 0.2f, 0.2f );
92  clearValues[1].depthStencil = vk::ClearDepthStencilValue( 1.0f, 0 );
93 
94  commandBuffer.begin( vk::CommandBufferBeginInfo( vk::CommandBufferUsageFlags() ) );
95 
96  vk::RenderPassBeginInfo renderPassBeginInfo(
97  renderPass, framebuffers[currentBuffer.value], vk::Rect2D( vk::Offset2D( 0, 0 ), surfaceData.extent ), clearValues );
98  commandBuffer.beginRenderPass( renderPassBeginInfo, vk::SubpassContents::eInline );
99 
100  commandBuffer.bindVertexBuffers( 0, vertexBuffer, { 0 } );
101 
102  commandBuffer.endRenderPass();
103  commandBuffer.end();
104  vk::su::submitAndWait( device, graphicsQueue, commandBuffer );
105 
106  device.destroySemaphore( imageAcquiredSemaphore );
107  device.destroyBuffer( vertexBuffer ); // destroy the buffer before the bound device memory to prevent some validation layer warning
108  device.freeMemory( deviceMemory );
109 
110  /* VULKAN_KEY_END */
111 
112  swapChainData.clear( device );
113  depthBufferData.clear( device );
114  for ( auto const & framebuffer : framebuffers )
115  {
116  device.destroyFramebuffer( framebuffer );
117  }
118  device.destroyRenderPass( renderPass );
119  device.freeCommandBuffers( commandPool, commandBuffer );
120  device.destroyCommandPool( commandPool );
121  device.destroy();
122  instance.destroySurfaceKHR( surfaceData.surface );
123 #if !defined( NDEBUG )
124  instance.destroyDebugUtilsMessengerEXT( debugUtilsMessenger );
125 #endif
126  instance.destroy();
127  }
128  catch ( vk::SystemError & err )
129  {
130  std::cout << "vk::SystemError: " << err.what() << std::endl;
131  exit( -1 );
132  }
133  catch ( std::exception & err )
134  {
135  std::cout << "std::exception: " << err.what() << std::endl;
136  exit( -1 );
137  }
138  catch ( ... )
139  {
140  std::cout << "unknown error\n";
141  exit( -1 );
142  }
143  return 0;
144 }
