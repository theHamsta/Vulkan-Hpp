53 
54  vk::CommandPool commandPool = device.createCommandPool( { {}, graphicsAndPresentQueueFamilyIndex.first } );
55  vk::CommandBuffer commandBuffer =
56  device.allocateCommandBuffers( vk::CommandBufferAllocateInfo( commandPool, vk::CommandBufferLevel::ePrimary, 1 ) ).front();

59  vk::Queue presentQueue = device.getQueue( graphicsAndPresentQueueFamilyIndex.second, 0 );

67  graphicsAndPresentQueueFamilyIndex.first,

73  commandBuffer.begin( vk::CommandBufferBeginInfo() );
74  textureData.setImage( device, commandBuffer, vk::su::CheckerboardImageGenerator() );
75 
76  vk::su::BufferData uniformBufferData( physicalDevice, device, sizeof( glm::mat4x4 ), vk::BufferUsageFlagBits::eUniformBuffer );
77  glm::mat4x4 mvpcMatrix = vk::su::createModelViewProjectionClipMatrix( surfaceData.extent );
78  vk::su::copyToDevice( device, uniformBufferData.deviceMemory, mvpcMatrix );

81  vk::su::createDescriptorSetLayout( device,
82  { { vk::DescriptorType::eUniformBuffer, 1, vk::ShaderStageFlagBits::eVertex },
83  { vk::DescriptorType::eCombinedImageSampler, 1, vk::ShaderStageFlagBits::eFragment } } );
84  vk::PipelineLayout pipelineLayout = device.createPipelineLayout( vk::PipelineLayoutCreateInfo( vk::PipelineLayoutCreateFlags(), descriptorSetLayout ) );
85 
86  vk::RenderPass renderPass = vk::su::createRenderPass(
87  device, vk::su::pickSurfaceFormat( physicalDevice.getSurfaceFormatsKHR( surfaceData.surface ) ).format, depthBufferData.format );

89  glslang::InitializeProcess();

91  vk::ShaderModule fragmentShaderModule = vk::su::createShaderModule( device, vk::ShaderStageFlagBits::eFragment, fragmentShaderText_T_C );
92  glslang::FinalizeProcess();

94  std::vector<vk::Framebuffer> framebuffers =
95  vk::su::createFramebuffers( device, renderPass, swapChainData.imageViews, depthBufferData.imageView, surfaceData.extent );
96 
97  vk::su::BufferData vertexBufferData( physicalDevice, device, sizeof( texturedCubeData ), vk::BufferUsageFlagBits::eVertexBuffer );
98  vk::su::copyToDevice( device, vertexBufferData.deviceMemory, texturedCubeData, sizeof( texturedCubeData ) / sizeof( texturedCubeData[0] ) );
99

101  vk::su::createDescriptorPool( device, { { vk::DescriptorType::eUniformBuffer, 1 }, { vk::DescriptorType::eCombinedImageSampler, 1 } } );

103  vk::DescriptorSet descriptorSet = device.allocateDescriptorSets( descriptorSetAllocateInfo ).front();

105  vk::su::updateDescriptorSets( device, descriptorSet, { { vk::DescriptorType::eUniformBuffer, uniformBufferData.buffer, VK_WHOLE_SIZE, {} } }, textureData );

107  vk::PipelineCache pipelineCache = device.createPipelineCache( vk::PipelineCacheCreateInfo() );

111  // Create two pipelines.
112  //
113  // First pipeline has VK_PIPELINE_CREATE_ALLOW_DERIVATIVES_BIT set.
114  // Second pipeline has a modified fragment shader and sets the VK_PIPELINE_CREATE_DERIVATIVE_BIT flag.
115 
116  std::array<vk::PipelineShaderStageCreateInfo, 2> pipelineShaderStageCreateInfos = {
117  vk::PipelineShaderStageCreateInfo( vk::PipelineShaderStageCreateFlags(), vk::ShaderStageFlagBits::eVertex, vertexShaderModule, "main" ),
118  vk::PipelineShaderStageCreateInfo( vk::PipelineShaderStageCreateFlags(), vk::ShaderStageFlagBits::eFragment, fragmentShaderModule, "main" )
119  };
120 
121  vk::VertexInputBindingDescription vertexInputBindingDescription( 0, sizeof( texturedCubeData[0] ) );

123  vk::VertexInputAttributeDescription( 0, 0, vk::Format::eR32G32B32A32Sfloat, 0 ),
124  vk::VertexInputAttributeDescription( 1, 0, vk::Format::eR32G32B32A32Sfloat, 16 )
125  };
126  vk::PipelineVertexInputStateCreateInfo pipelineVertexInputStateCreateInfo(
127  vk::PipelineVertexInputStateCreateFlags(), vertexInputBindingDescription, vertexInputAttributeDescriptions );
128 
129  vk::PipelineInputAssemblyStateCreateInfo pipelineInputAssemblyStateCreateInfo( vk::PipelineInputAssemblyStateCreateFlags(),
130  vk::PrimitiveTopology::eTriangleList );
131 
132  vk::PipelineViewportStateCreateInfo pipelineViewportStateCreateInfo( vk::PipelineViewportStateCreateFlags(), 1, nullptr, 1, nullptr );
133 
134  vk::PipelineRasterizationStateCreateInfo pipelineRasterizationStateCreateInfo( vk::PipelineRasterizationStateCreateFlags(),
135  false,
136  false,
137  vk::PolygonMode::eFill,
138  vk::CullModeFlagBits::eBack,
139  vk::FrontFace::eClockwise,
140  false,
141  0.0f,
142  0.0f,
143  0.0f,
144  1.0f );
145 
146  vk::PipelineMultisampleStateCreateInfo pipelineMultisampleStateCreateInfo( {}, vk::SampleCountFlagBits::e1 );
147 
148  vk::StencilOpState stencilOpState( vk::StencilOp::eKeep, vk::StencilOp::eKeep, vk::StencilOp::eKeep, vk::CompareOp::eAlways );
149  vk::PipelineDepthStencilStateCreateInfo pipelineDepthStencilStateCreateInfo(
150  vk::PipelineDepthStencilStateCreateFlags(), true, true, vk::CompareOp::eLessOrEqual, false, false, stencilOpState, stencilOpState );
151 
152  vk::ColorComponentFlags colorComponentFlags( vk::ColorComponentFlagBits::eR | vk::ColorComponentFlagBits::eG | vk::ColorComponentFlagBits::eB |
153  vk::ColorComponentFlagBits::eA );

155  vk::BlendFactor::eZero,
156  vk::BlendFactor::eZero,
157  vk::BlendOp::eAdd,
158  vk::BlendFactor::eZero,
159  vk::BlendFactor::eZero,
160  vk::BlendOp::eAdd,
161  colorComponentFlags );
162  vk::PipelineColorBlendStateCreateInfo pipelineColorBlendStateCreateInfo(
163  vk::PipelineColorBlendStateCreateFlags(), false, vk::LogicOp::eNoOp, pipelineColorBlendAttachmentState, { { 1.0f, 1.0f, 1.0f, 1.0f } } );
164 
165  std::array<vk::DynamicState, 2> dynamicStates = { vk::DynamicState::eViewport, vk::DynamicState::eScissor };
166  vk::PipelineDynamicStateCreateInfo pipelineDynamicStateCreateInfo( vk::PipelineDynamicStateCreateFlags(), dynamicStates );
167 
168  vk::GraphicsPipelineCreateInfo graphicsPipelineCreateInfo( vk::PipelineCreateFlagBits::eAllowDerivatives,
169  pipelineShaderStageCreateInfos,
170  &pipelineVertexInputStateCreateInfo,
171  &pipelineInputAssemblyStateCreateInfo,
172  nullptr,
173  &pipelineViewportStateCreateInfo,
174  &pipelineRasterizationStateCreateInfo,

179  pipelineLayout,

182  vk::Pipeline basePipeline;
183  vk::ResultValue<vk::Pipeline> rvPipeline = device.createGraphicsPipeline( pipelineCache, graphicsPipelineCreateInfo );
184  switch ( rvPipeline.result )
185  {
186  case vk::Result::eSuccess: basePipeline = std::move( rvPipeline.value ); break;
187  case vk::Result::ePipelineCompileRequiredEXT:
188  // something meaningfull here
189  break;
190  default: assert( false ); // should never happen
191  }

193  // Now create the derivative pipeline, using a different fragment shader
194  // This shader will shade the cube faces with interpolated colors
195  const std::string fragmentShaderText_T_C_2 = R"(
196 #version 450
197 
198 layout (location = 0) in vec2 inTexCoord;
199 
200 layout (location = 0) out vec4 outColor;
201 
202 void main()
203 {
204  outColor = vec4(inTexCoord.x, inTexCoord.y, 1.0f - inTexCoord.x - inTexCoord.y, 1.0f);
205 }
206 )";
207 
208  // Convert GLSL to SPIR-V
209  glslang::InitializeProcess();
210  vk::ShaderModule fragmentShaderModule2 = vk::su::createShaderModule( device, vk::ShaderStageFlagBits::eFragment, fragmentShaderText_T_C_2 );
211  glslang::FinalizeProcess();
212 
213  // Modify pipeline info to reflect derivation
214  pipelineShaderStageCreateInfos[1] =
215  vk::PipelineShaderStageCreateInfo( vk::PipelineShaderStageCreateFlags(), vk::ShaderStageFlagBits::eFragment, fragmentShaderModule2, "main" );
216  graphicsPipelineCreateInfo.flags = vk::PipelineCreateFlagBits::eDerivative;
217  graphicsPipelineCreateInfo.basePipelineHandle = basePipeline;
218  graphicsPipelineCreateInfo.basePipelineIndex = -1;
219 
220  // And create the derived pipeline
221  vk::Pipeline derivedPipeline;
222  rvPipeline = device.createGraphicsPipeline( pipelineCache, graphicsPipelineCreateInfo );
223  switch ( rvPipeline.result )
224  {
225  case vk::Result::eSuccess: derivedPipeline = std::move( rvPipeline.value ); break;
226  case vk::Result::ePipelineCompileRequiredEXT:
227  // something meaningfull here
228  break;
229  default: assert( false ); // should never happen
230  }

234  vk::Semaphore imageAcquiredSemaphore = device.createSemaphore( vk::SemaphoreCreateInfo( vk::SemaphoreCreateFlags() ) );
235 
236  // Get the index of the next available swapchain image
237  vk::ResultValue<uint32_t> currentBuffer = device.acquireNextImageKHR( swapChainData.swapChain, UINT64_MAX, imageAcquiredSemaphore, nullptr );
238  assert( currentBuffer.result == vk::Result::eSuccess );
239  assert( currentBuffer.value < framebuffers.size() );
240 
241  std::array<vk::ClearValue, 2> clearValues;
242  clearValues[0].color = vk::ClearColorValue( 0.2f, 0.2f, 0.2f, 0.2f );
243  clearValues[1].depthStencil = vk::ClearDepthStencilValue( 1.0f, 0 );
244 
245  commandBuffer.beginRenderPass(
246  vk::RenderPassBeginInfo( renderPass, framebuffers[currentBuffer.value], vk::Rect2D( vk::Offset2D(), surfaceData.extent ), clearValues ),
247  vk::SubpassContents::eInline );
248  commandBuffer.bindPipeline( vk::PipelineBindPoint::eGraphics, derivedPipeline );
249  commandBuffer.bindDescriptorSets( vk::PipelineBindPoint::eGraphics, pipelineLayout, 0, descriptorSet, {} );
250 
251  commandBuffer.bindVertexBuffers( 0, vertexBufferData.buffer, { 0 } );
252  commandBuffer.setViewport(
253  0, vk::Viewport( 0.0f, 0.0f, static_cast<float>( surfaceData.extent.width ), static_cast<float>( surfaceData.extent.height ), 0.0f, 1.0f ) );
254  commandBuffer.setScissor( 0, vk::Rect2D( vk::Offset2D( 0, 0 ), surfaceData.extent ) );
255 
256  commandBuffer.draw( 12 * 3, 1, 0, 0 );
257  commandBuffer.endRenderPass();
258  commandBuffer.end();
259 
260  vk::Fence drawFence = device.createFence( vk::FenceCreateInfo() );
261 
262  vk::PipelineStageFlags waitDestinationStageMask( vk::PipelineStageFlagBits::eColorAttachmentOutput );
263  vk::SubmitInfo submitInfo( imageAcquiredSemaphore, waitDestinationStageMask, commandBuffer );
264  graphicsQueue.submit( submitInfo, drawFence );
265 
266  while ( vk::Result::eTimeout == device.waitForFences( drawFence, VK_TRUE, vk::su::FenceTimeout ) )
267  ;
268 
269  vk::Result result = presentQueue.presentKHR( vk::PresentInfoKHR( {}, swapChainData.swapChain, currentBuffer.value ) );
270  switch ( result )
271  {
272  case vk::Result::eSuccess: break;
273  case vk::Result::eSuboptimalKHR: std::cout << "vk::Queue::presentKHR returned vk::Result::eSuboptimalKHR !\n"; break;
274  default: assert( false ); // an unexpected result is returned !
275  }
276  std::this_thread::sleep_for( std::chrono::milliseconds( 1000 ) );
277 
278  device.destroyFence( drawFence );
279  device.destroySemaphore( imageAcquiredSemaphore );
280  device.destroyShaderModule( fragmentShaderModule2 );
281  device.destroyPipeline( derivedPipeline );
282  device.destroyPipeline( basePipeline );
283  device.destroyPipelineCache( pipelineCache );
284  device.freeDescriptorSets( descriptorPool, descriptorSet );
285  device.destroyDescriptorPool( descriptorPool );
286  vertexBufferData.clear( device );
287  for ( auto framebuffer : framebuffers )
288  {
289  device.destroyFramebuffer( framebuffer );
290  }
291  device.destroyShaderModule( fragmentShaderModule );
292  device.destroyShaderModule( vertexShaderModule );
293  device.destroyRenderPass( renderPass );
294  device.destroyPipelineLayout( pipelineLayout );
295  device.destroyDescriptorSetLayout( descriptorSetLayout );
296  uniformBufferData.clear( device );
297  textureData.clear( device );
298  depthBufferData.clear( device );
299  swapChainData.clear( device );
300  device.freeCommandBuffers( commandPool, commandBuffer );
301  device.destroyCommandPool( commandPool );
302  device.destroy();
303  instance.destroySurfaceKHR( surfaceData.surface );
304 #if !defined( NDEBUG )
305  instance.destroyDebugUtilsMessengerEXT( debugUtilsMessenger );
306 #endif
307  instance.destroy();
308  }
309  catch ( vk::SystemError & err )
310  {
311  std::cout << "vk::SystemError: " << err.what() << std::endl;
312  exit( -1 );
313  }
314  catch ( std::exception & err )
315  {
316  std::cout << "std::exception: " << err.what() << std::endl;
317  exit( -1 );
318  }
319  catch ( ... )
320  {
321  std::cout << "unknown error\n";
322  exit( -1 );
323  }
324  return 0;
