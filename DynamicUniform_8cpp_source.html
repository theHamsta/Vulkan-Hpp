37 static char const * AppName = "DynamicUniform";
38 static char const * EngineName = "Vulkan.hpp";
39 
40 int main( int /*argc*/, char ** /*argv*/ )
41 {
42  try
43  {
44  vk::Instance instance = vk::su::createInstance( AppName, EngineName, {}, vk::su::getInstanceExtensions() );
45 #if !defined( NDEBUG )
46  vk::DebugUtilsMessengerEXT debugUtilsMessenger = instance.createDebugUtilsMessengerEXT( vk::su::makeDebugUtilsMessengerCreateInfoEXT() );
47 #endif
48 
49  vk::PhysicalDevice physicalDevice = instance.enumeratePhysicalDevices().front();
50 
51  vk::su::SurfaceData surfaceData( instance, AppName, vk::Extent2D( 500, 500 ) );
52 
53  std::pair<uint32_t, uint32_t> graphicsAndPresentQueueFamilyIndex = vk::su::findGraphicsAndPresentQueueFamilyIndex( physicalDevice, surfaceData.surface );
54  vk::Device device = vk::su::createDevice( physicalDevice, graphicsAndPresentQueueFamilyIndex.first, vk::su::getDeviceExtensions() );
55 
56  vk::CommandPool commandPool = device.createCommandPool( { {}, graphicsAndPresentQueueFamilyIndex.first } );
57  vk::CommandBuffer commandBuffer =
58  device.allocateCommandBuffers( vk::CommandBufferAllocateInfo( commandPool, vk::CommandBufferLevel::ePrimary, 1 ) ).front();
59 
60  vk::Queue graphicsQueue = device.getQueue( graphicsAndPresentQueueFamilyIndex.first, 0 );
61  vk::Queue presentQueue = device.getQueue( graphicsAndPresentQueueFamilyIndex.second, 0 );
62 
63  vk::su::SwapChainData swapChainData( physicalDevice,
64  device,
65  surfaceData.surface,
66  surfaceData.extent,
67  vk::ImageUsageFlagBits::eColorAttachment | vk::ImageUsageFlagBits::eTransferSrc,
68  {},
69  graphicsAndPresentQueueFamilyIndex.first,
70  graphicsAndPresentQueueFamilyIndex.second );
71 
72  vk::su::DepthBufferData depthBufferData( physicalDevice, device, vk::Format::eD16Unorm, surfaceData.extent );
73 
74  vk::RenderPass renderPass = vk::su::createRenderPass(
75  device, vk::su::pickSurfaceFormat( physicalDevice.getSurfaceFormatsKHR( surfaceData.surface ) ).format, depthBufferData.format );
76 
77  glslang::InitializeProcess();
78  vk::ShaderModule vertexShaderModule = vk::su::createShaderModule( device, vk::ShaderStageFlagBits::eVertex, vertexShaderText_PC_C );
79  vk::ShaderModule fragmentShaderModule = vk::su::createShaderModule( device, vk::ShaderStageFlagBits::eFragment, fragmentShaderText_C_C );
80  glslang::FinalizeProcess();
81 
82  std::vector<vk::Framebuffer> framebuffers =
83  vk::su::createFramebuffers( device, renderPass, swapChainData.imageViews, depthBufferData.imageView, surfaceData.extent );
84 
85  vk::su::BufferData vertexBufferData( physicalDevice, device, sizeof( coloredCubeData ), vk::BufferUsageFlagBits::eVertexBuffer );
86  vk::su::copyToDevice( device, vertexBufferData.deviceMemory, coloredCubeData, sizeof( coloredCubeData ) / sizeof( coloredCubeData[0] ) );
87 
88  /* VULKAN_KEY_START */
89 
90  vk::PhysicalDeviceLimits limits = physicalDevice.getProperties().limits;
91  if ( limits.maxDescriptorSetUniformBuffersDynamic < 1 )
92  {
93  std::cout << "No dynamic uniform buffers supported\n";
94  exit( -1 );
95  }
96 
97  /* Set up uniform buffer with 2 transform matrices in it */
98  glm::mat4x4 mvpcs[2];
99  glm::mat4x4 model = glm::mat4x4( 1.0f );
100  glm::mat4x4 view = glm::lookAt( glm::vec3( 0.0f, 3.0f, -10.0f ), glm::vec3( 0.0f, 0.0f, 0.0f ), glm::vec3( 0.0f, -1.0f, 0.0f ) );
101  glm::mat4x4 projection = glm::perspective( glm::radians( 45.0f ), 1.0f, 0.1f, 100.0f );
102  // clang-format off
103  glm::mat4x4 clip = glm::mat4x4( 1.0f, 0.0f, 0.0f, 0.0f,
104  0.0f, -1.0f, 0.0f, 0.0f,
105  0.0f, 0.0f, 0.5f, 0.0f,
106  0.0f, 0.0f, 0.5f, 1.0f ); // vulkan clip space has inverted y and half z !
107  // clang-format on
108  mvpcs[0] = clip * projection * view * model;
109 
110  model = glm::translate( model, glm::vec3( -1.5f, 1.5f, -1.5f ) );
111  mvpcs[1] = clip * projection * view * model;
112 
113  vk::DeviceSize bufferSize = sizeof( glm::mat4x4 );
114  if ( limits.minUniformBufferOffsetAlignment )
115  {
116  bufferSize = ( bufferSize + limits.minUniformBufferOffsetAlignment - 1 ) & ~( limits.minUniformBufferOffsetAlignment - 1 );
117  }
118 
119  vk::su::BufferData uniformBufferData( physicalDevice, device, 2 * bufferSize, vk::BufferUsageFlagBits::eUniformBuffer );
120  vk::su::copyToDevice( device, uniformBufferData.deviceMemory, mvpcs, 2, bufferSize );
121 
122  // create a DescriptorSetLayout with vk::DescriptorType::eUniformBufferDynamic
123  vk::DescriptorSetLayout descriptorSetLayout =
124  vk::su::createDescriptorSetLayout( device, { { vk::DescriptorType::eUniformBufferDynamic, 1, vk::ShaderStageFlagBits::eVertex } } );
125  vk::PipelineLayout pipelineLayout = device.createPipelineLayout( vk::PipelineLayoutCreateInfo( vk::PipelineLayoutCreateFlags(), descriptorSetLayout ) );
126 
127  // create a DescriptorPool with vk::DescriptorType::eUniformBufferDynamic
128  vk::DescriptorPool descriptorPool = vk::su::createDescriptorPool( device, { { vk::DescriptorType::eUniformBufferDynamic, 1 } } );
129  vk::DescriptorSetAllocateInfo descriptorSetAllocateInfo( descriptorPool, descriptorSetLayout );
130  vk::DescriptorSet descriptorSet = device.allocateDescriptorSets( descriptorSetAllocateInfo ).front();
131 
132  vk::su::updateDescriptorSets( device, descriptorSet, { { vk::DescriptorType::eUniformBufferDynamic, uniformBufferData.buffer, bufferSize, {} } }, {} );
133 
134  vk::PipelineCache pipelineCache = device.createPipelineCache( vk::PipelineCacheCreateInfo() );
135  vk::Pipeline graphicsPipeline = vk::su::createGraphicsPipeline( device,
136  pipelineCache,
137  std::make_pair( vertexShaderModule, nullptr ),
138  std::make_pair( fragmentShaderModule, nullptr ),
139  sizeof( coloredCubeData[0] ),
140  { { vk::Format::eR32G32B32A32Sfloat, 0 }, { vk::Format::eR32G32B32A32Sfloat, 16 } },
141  vk::FrontFace::eClockwise,
142  true,
143  pipelineLayout,
144  renderPass );
145  // Get the index of the next available swapchain image:
146  vk::Semaphore imageAcquiredSemaphore = device.createSemaphore( vk::SemaphoreCreateInfo() );
147  vk::ResultValue<uint32_t> currentBuffer = device.acquireNextImageKHR( swapChainData.swapChain, vk::su::FenceTimeout, imageAcquiredSemaphore, nullptr );
148  assert( currentBuffer.result == vk::Result::eSuccess );
149  assert( currentBuffer.value < framebuffers.size() );
150 
151  commandBuffer.begin( vk::CommandBufferBeginInfo( vk::CommandBufferUsageFlags() ) );
152 
153  std::array<vk::ClearValue, 2> clearValues;
154  clearValues[0].color = vk::ClearColorValue( 0.2f, 0.2f, 0.2f, 0.2f );
155  clearValues[1].depthStencil = vk::ClearDepthStencilValue( 1.0f, 0 );
156  vk::RenderPassBeginInfo renderPassBeginInfo(
157  renderPass, framebuffers[currentBuffer.value], vk::Rect2D( vk::Offset2D( 0, 0 ), surfaceData.extent ), clearValues );
158  commandBuffer.beginRenderPass( renderPassBeginInfo, vk::SubpassContents::eInline );
159  commandBuffer.bindPipeline( vk::PipelineBindPoint::eGraphics, graphicsPipeline );
160 
161  commandBuffer.setViewport(
162  0, vk::Viewport( 0.0f, 0.0f, static_cast<float>( surfaceData.extent.width ), static_cast<float>( surfaceData.extent.height ), 0.0f, 1.0f ) );
163  commandBuffer.setScissor( 0, vk::Rect2D( vk::Offset2D( 0, 0 ), surfaceData.extent ) );
164 
165  /* The first draw should use the first matrix in the buffer */
166  uint32_t dynamicOffset = 0;
167  commandBuffer.bindDescriptorSets( vk::PipelineBindPoint::eGraphics, pipelineLayout, 0, descriptorSet, dynamicOffset );
168 
169  commandBuffer.bindVertexBuffers( 0, vertexBufferData.buffer, { 0 } );
170  commandBuffer.draw( 12 * 3, 1, 0, 0 );
171 
172  // the second draw should use the second matrix in the buffer;
173  dynamicOffset = (uint32_t)bufferSize;
174  commandBuffer.bindDescriptorSets( vk::PipelineBindPoint::eGraphics, pipelineLayout, 0, descriptorSet, dynamicOffset );
175  commandBuffer.draw( 12 * 3, 1, 0, 0 );
176 
177  commandBuffer.endRenderPass();
178  commandBuffer.end();
179 
180  vk::Fence drawFence = device.createFence( vk::FenceCreateInfo() );
181 
182  vk::PipelineStageFlags waitDestinationStageMask( vk::PipelineStageFlagBits::eColorAttachmentOutput );
183  vk::SubmitInfo submitInfo( imageAcquiredSemaphore, waitDestinationStageMask, commandBuffer );
184  graphicsQueue.submit( submitInfo, drawFence );
185 
186  while ( vk::Result::eTimeout == device.waitForFences( drawFence, VK_TRUE, vk::su::FenceTimeout ) )
187  ;
188 
189  vk::Result result = presentQueue.presentKHR( vk::PresentInfoKHR( {}, swapChainData.swapChain, currentBuffer.value ) );
190  switch ( result )
191  {
192  case vk::Result::eSuccess: break;
193  case vk::Result::eSuboptimalKHR: std::cout << "vk::Queue::presentKHR returned vk::Result::eSuboptimalKHR !\n"; break;
194  default: assert( false ); // an unexpected result is returned !
195  }
196  std::this_thread::sleep_for( std::chrono::milliseconds( 1000 ) );
197 
198  /* VULKAN_KEY_END */
199 
200  device.waitIdle();
201 
202  device.destroyFence( drawFence );
203  device.destroySemaphore( imageAcquiredSemaphore );
204  device.destroyPipeline( graphicsPipeline );
205  device.destroyPipelineCache( pipelineCache );
206  device.freeDescriptorSets( descriptorPool, descriptorSet );
207  device.destroyDescriptorPool( descriptorPool );
208  device.destroyPipelineLayout( pipelineLayout );
209  device.destroyDescriptorSetLayout( descriptorSetLayout );
210  uniformBufferData.clear( device );
211  vertexBufferData.clear( device );
212  for ( auto framebuffer : framebuffers )
213  {
214  device.destroyFramebuffer( framebuffer );
215  }
216  device.destroyShaderModule( fragmentShaderModule );
217  device.destroyShaderModule( vertexShaderModule );
218  device.destroyRenderPass( renderPass );
219  depthBufferData.clear( device );
220  swapChainData.clear( device );
221  device.freeCommandBuffers( commandPool, commandBuffer );
222  device.destroyCommandPool( commandPool );
223  device.destroy();
224  instance.destroySurfaceKHR( surfaceData.surface );
225 #if !defined( NDEBUG )
226  instance.destroyDebugUtilsMessengerEXT( debugUtilsMessenger );
227 #endif
228  instance.destroy();
229  }
230  catch ( vk::SystemError & err )
231  {
232  std::cout << "vk::SystemError: " << err.what() << std::endl;
233  exit( -1 );
234  }
235  catch ( std::exception & err )
236  {
237  std::cout << "std::exception: " << err.what() << std::endl;
238  exit( -1 );
239  }
240  catch ( ... )
241  {
242  std::cout << "unknown error\n";
243  exit( -1 );
244  }
245  return 0;
246 }
